Example I: Simple complete graph using popy only

[7]:

import popy


class FullyConnected(popy.MagicLocation):
    pass


model = popy.Model()
creator = popy.Creator(model)
inspector = popy.NetworkInspector(model)

for _ in range(5):
    popy.Agent(model)

creator.create_locations(location_classes=[FullyConnected])

Example II: Complete graph with one location for each bilateral relation using networkx

[10]:

import networkx as nx


class FullyConnected(popy.MagicLocation):
    nxgraph = nx.complete_graph(n=5)


model = popy.Model()
creator = popy.Creator(model)
inspector = popy.NetworkInspector(model)

for _ in range(5):
    popy.Agent(model)

creator.create_locations(location_classes=[FullyConnected])
